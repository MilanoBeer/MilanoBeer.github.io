# [BJ]7562 나이트의 이동

✔️ 나이트가 갈 수 있는 8방위를 반복적으로 탐색하면서 Queue에 담는다.
    
    해당 지점까지의 이동횟수는 map의 각 지점에  value값으로 갱신해서 저장해둔다. 
    
    현재지점에서 8방위탐색으로 통해 다른 지점으로 갈 수 있다면, map에서 해당 위치에 +1 을 한다. 
    

```java
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class BJ_7562_나이트의이동 {
/*
BJ_7562_나이트의 이동   
*/	
	
	static int[][] dir = {
			{-2, 1}, {-1, 2}, {1, 2}, {2, 1},
			{2, -1}, {1, -2}, {-1, -2}, {-2, -1}
	};
	
	static int T; 
	static int I; 
	static int[] cur = new int[2]; 
	static int[] move = new int[2]; 
	static int[][] map ; 
	static int cnt = 0; 
	
	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);
		T = sc.nextInt(); 
		
		for (int t = 0; t < T; t++) {
			I = sc.nextInt(); 
			map = new int[I][I]; 
			
			// input position data 
			cur[0] = sc.nextInt(); 
			cur[1] = sc.nextInt(); 
			
			move[0] = sc.nextInt(); 
			move[1] = sc.nextInt(); 
			
			cnt = 0; 
			bfs(); 
			
		}
	}

	private static void bfs() {
		Queue<int[]> q = new LinkedList<>();
		boolean[][] V = new boolean[I][I]; 
		
		q.offer(new int[] {cur[0], cur[1]});
		V[cur[0]][cur[1]] = true; 
		
		while(!q.isEmpty()) {
			int[] curPos = q.poll(); 
			int cr = curPos[0];
			int cc = curPos[1];
			
			if(cr == move[0] && cc == move[1]) {
				System.out.println(map[cr][cc]);
				break; 
			}
			
			// 8방위 검사 
			for (int d = 0; d < 8; d++) {
				int nr = cr + dir[d][0];
				int nc = cc + dir[d][1];
				
				// 경계검사
				if(nr < 0 || nr >= I || nc < 0 || nc >= I) continue; 
				
				// 방문 검사 
				if(V[nr][nc]) continue; 
				
				// 아직 도착전이라면, 큐에 추가 
				q.offer(new int[] {nr, nc});
				map[nr][nc] = map[cr][cc] +1; // 이전위치까지의 이동횟수 + 1 을 해서 대입
				V[nr][nc] = true; 
			}
		} // end : while is Empty 
	}
}
```
