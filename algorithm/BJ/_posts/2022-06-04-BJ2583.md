# ❗BJ_2583_영역구하기

💡2차원 배열을 좌표평면 형식으로 다루는 방법 

💡ArrayList 정렬하는 방법

✔️ 탐색하러 갈 때, 빈 영역이면서 아직 방문하지 않은 영역인지 같이 검사하기

```java
import java.util.ArrayList;
import java.util.Collections;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class BJ_2583_영역구하기 {
	/*
	BJ_2583_영역구하기 
	 */
	/*
	5 7 3
	0 2 4 4
	1 1 2 5
	4 0 6 2
	 */
	static int M, N, K; // 행, 렬, 직사각형 개수
	static int[][] map; 
	static boolean[][] V; 
	static int[][] dir = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}}; // 상하좌우 

	static int cnt = 0; 
	static int area = 0; 
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		M = sc.nextInt(); 
		N= sc.nextInt(); 
		K = sc.nextInt();

		// 행렬이 아니라 좌표평면으로 다뤄야 할 때, map에 적절히 값 배치할 수 있기**
		map = new int[N][M]; // N, M으로 입력받는다.
		V = new boolean[N][M];

		// 왼쪽 아랫쪽 지점 , 오른쪽 상담지점 
		int x1, y1, x2, y2; 
		for (int i = 0; i < K; i++) {
			x1 = sc.nextInt(); y1 = sc.nextInt();
			x2 = sc.nextInt(); y2 = sc.nextInt(); 

			// 각 직사각형 영역알아내서, 1로 바꿈  
			for (int w = x1; w < x2; w++) {
				for (int h = y1; h < y2; h++) {
					map[w][h] = 1; 
				}
			}
		} // end

		ArrayList<Integer> list = new ArrayList<>(); // 각 영역의 넓이값 추가할 list  
		
		// 영역을 돌다가, 빈 영역이면서 && 방문한적이 없으면 탐색 시작
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				if(map[i][j] == 0 && !V[i][j]) {
					area = 1; // 첫 영역의 넓이 1 부터 시작
					go(i, j);
					cnt++; // 영역 갯수 + 1
					list.add(area); //해당 영역의 넓이 저장
				}
			}
		}
		// ArrayList정렬하기
		Collections.sort(list);
		
		System.out.println(cnt);
		for (int i = 0; i < list.size(); i++) {
			System.out.print(list.get(i) + " ");
		}
		System.out.println();
	}

	// BFS
	private static void go(int x, int y) {
		// 빈영역에서 출발 
		Queue<int[]> q = new LinkedList<>(); 
		q.offer(new int[] {x, y}); 
		V[x][y] = true; 

		while(!q.isEmpty()) {
			int[] curr = q.poll(); 

			for (int d = 0; d < 4; d++) {
				int nx = curr[0] + dir[d][0];
				int ny = curr[1] + dir[d][1];

				//경계검사
				if(nx < 0 || nx >= N || ny <0 || ny >= M)
					continue; 

				// 방문체크
				if(V[nx][ny])
					continue; 

				// q.에 offer , 넓이 세기 
				if(map[nx][ny] == 0)
				{
					q.offer(new int[] {nx, ny});
					V[nx][ny] = true; 
					area++;
				}
			}
		} // end : while isEmpty 
	}

}
```
