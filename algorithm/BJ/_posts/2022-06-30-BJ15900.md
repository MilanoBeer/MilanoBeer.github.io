# [BJ] 15900 나무탈출

 ***🏹 문제 링크***

[15900번: 나무 탈출](https://www.acmicpc.net/problem/15900)

✏️ ***풀기 전 메모*** 

- 인접리스트
- Solution>
    - BFS
    - 리프노드 → 루트노드까지의 거리의 총합이 짝수라면 패배, 홀수라면 승리
    - 리프노드에 대해서만, 그 합을 구해야함
    - 리프노드 구별하는 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

/* BJ_15900_나무탈출 */
public class BJ_15900_나무탈출 {

	static int N; 
	static ArrayList<ArrayList<Integer>> list = new ArrayList<>();
	static boolean[] V ; 
	static int cnt = 0; 
	static int Ans = 0; 
	public static void main(String[] args) throws IOException {
		// Scanner 시간초과남
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in)); 
		StringTokenizer st = new StringTokenizer(br.readLine());

		N = Integer.parseInt(st.nextToken()); 
		V = new boolean[N+1]; 

		for (int i = 0; i <= N; i++) {
			list.add(new ArrayList<>());
		}

		for (int i = 1; i < N; i++) {
			st = new StringTokenizer(br.readLine()); 
			int from = Integer.parseInt(st.nextToken()); 
			int to = Integer.parseInt(st.nextToken());

			list.get(from).add(to); 
			list.get(to).add(from); 
		}

		// 루트노드 1부터 시작 
		find(1); 

		// 짝수면 성원이 패배 / 홀수면 승리 
		if(Ans % 2 == 0) {
			System.out.println("No");
		}else {
			System.out.println("Yes");
		}
		br.close(); 
	}

	static boolean FLAG = false; 
	private static void find(int start) {
		Queue<int[]> q = new LinkedList<>(); 
		q.add(new int[] {start, 0}); 
		V[start] = true; 

		while(!q.isEmpty()) {
			FLAG = false; // 매번 초기화 
			
			int[] curr = q.poll(); 
			int curV = curr[0]; 
			int curCnt = curr[1];

			for (int i = 0; i < list.get(curV).size(); i++) {
		
				int tmp = list.get(curV).get(i); 
				if(V[tmp]) {
//					System.out.println("tmp: " + tmp);
					continue;
				}

				V[tmp] = true; 
				q.add(new int[] {tmp, curCnt+1}); 
				cnt++; 
				FLAG = true; 
			} 

			if(!FLAG) {
				Ans += curCnt;
			}
		}

	}

}
```
