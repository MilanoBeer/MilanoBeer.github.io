import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

/* BJ_15900_ë‚˜ë¬´íƒˆì¶œ */
public class BJ_15900_ë‚˜ë¬´íƒˆì¶œ {

	static int N; 
	static ArrayList<ArrayList<Integer>> list = new ArrayList<>();
	static boolean[] V ; 
	static int cnt = 0; 
	static int Ans = 0; 
	public static void main(String[] args) throws IOException {
		// Scanner ì‹œê°„ì´ˆê³¼ë‚¨
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in)); 
		StringTokenizer st = new StringTokenizer(br.readLine());

		N = Integer.parseInt(st.nextToken()); 
		V = new boolean[N+1]; 

		for (int i = 0; i <= N; i++) {
			list.add(new ArrayList<>());
		}

		for (int i = 1; i < N; i++) {
			st = new StringTokenizer(br.readLine()); 
			int from = Integer.parseInt(st.nextToken()); 
			int to = Integer.parseInt(st.nextToken());

			list.get(from).add(to); 
			list.get(to).add(from); 
		}

		// ë£¨íŠ¸ë…¸ë“œ 1ë¶€í„° ì‹œì‘ 
		find(1); 

		// ì§ìˆ˜ë©´ ì„±ì›ì´ íŒ¨ë°° / í™€ìˆ˜ë©´ ìŠ¹ë¦¬ 
		if(Ans % 2 == 0) {
			System.out.println("No");
		}else {
			System.out.println("Yes");
		}
		br.close(); 
	}

	static boolean FLAG = false; 
	private static void find(int start) {
		Queue<int[]> q = new LinkedList<>(); 
		q.add(new int[] {start, 0}); 
		V[start] = true; 

		while(!q.isEmpty()) {
			FLAG = false; // ë§¤ë²ˆ ì´ˆê¸°í™” 
			
			int[] curr = q.poll(); 
			int curV = curr[0]; 
			int curCnt = curr[1];

			for (int i = 0; i < list.get(curV).size(); i++) {
		
				int tmp = list.get(curV).get(i); 
				if(V[tmp]) {
//					System.out.println("tmp: " + tmp);
					continue;
				}

				V[tmp] = true; 
				q.add(new int[] {tmp, curCnt+1}); 
				cnt++; 
				FLAG = true; 
			} 

			if(!FLAG) {
				Ans += curCnt;
			}
		}

	}

}
